$(document).ready(function() {
    $(".tabs").hide();
        $(".tabs:first").show();
        $(".tabMenu li a").click(function () {
            var tablink = $(this).attr('href');
            $(".tabs").hide();
            $(tablink).show();
            $(this).parents(".tabMenu").find('li').removeClass("active");
            $(this).parents(".tabMenu li").addClass("active");
            return false;
    });

});

$(document).ready(function() {
    $(".tabs2").hide();
    $('.tabs2:first').show();
    $(".tabMenu2 li a").click(function () {
        var tablink = $(this).attr('href');
        $(".tabs2").fadeOut();
        $(tablink).stop().fadeIn();
        $(this).parents(".tabMenu2").find('li').removeClass("active");
        $(this).parents(".tabMenu2 li").addClass("active");
        return false;
    });

});
