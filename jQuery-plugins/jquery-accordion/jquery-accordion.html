$(document).ready(function(){
    $('.content').hide();
    $('.accord li a').click(function(){
    $(this).addClass('active');
    $(this).next('.content').slideDown();
    $(this).parents('li').siblings().find('a').removeClass('active');
    $(this).parents('li').siblings().find('.content').slideUp();
    return false;
    });
});
