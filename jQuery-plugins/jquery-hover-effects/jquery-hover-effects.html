$(document).ready(function() {
    $(".hoverDiv").css("opacity", "0");
        $(".fadeInEffect li").mouseenter(function(){
            $(this).find('.hoverDiv').stop().animate({opacity: "1"}, 500);
        });
        $(".fadeInEffect li").mouseleave(function(){
            $(this).find('.hoverDiv').stop().animate({opacity: "0"}, 500);
    });
});

Image Zoom effect on hover

                    

$(".imageHover img").mouseenter(function(){
    $(this).stop().animate({
        width :'120%',
        margin: '-10% 0 0 -10%'
        });
    });
    $(".imageHover img").mouseleave(function(){
        $(this).stop().animate({
        width :'100%',
        margin: '0'
    });
});
